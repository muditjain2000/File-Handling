import numpy as np 
import matplotlib.pyplot as plt

x = [-1*np.pi,-1*np.pi/4,-1*np.pi/2,0,np.pi/4,np.pi/2,np.pi]
x = np.array(x)

y_tan = np.tan(x)
y_cot = 1/np.tan(x)
y_sec = 1/np.cos(x)
y_csc = 1/np.sin(x)
plt.plot(x,y_tan,color='blue')
plt.plot(x,y_cot,color='red')
plt.plot(x,y_sec,color='green')
plt.plot(x,y_csc,color='yellow')
